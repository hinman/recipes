\recipe{Kelly's Apple Crips}
\ingred{
\begin{itemize}
\item Apples
\item Cinnamon
\item 1 cup sugar
\item 1 cup brown sugar
\item 2 cups oatmeal
\item $\frac{2}{3}$ cup flour
\item $\frac{2}{3}$ melted butter
\end{itemize}
}

\begin{enumerate}[{\color{blue} 1.}]
\addtolength{\itemindent}{2em}
 \item slice apples and layer in 9x13 inch pan
 \item add cinnamon to apples
 \item crumble remaining ingredients together
 \item top apples with crumble
 \item bake 350 degrees for 20 -- 25 minutes
\end{enumerate}
