\recipe{Vodka Gimlet}
\ingred{
\begin{itemize}
\item 2 oz. Vodka
\item 1 oz. fresh lime juice
\item $\frac{3}{4}$ oz. simple syrup
\end{itemize}
}

\begin{enumerate}[{\color{blue} 1.}]
\addtolength{\itemindent}{2em}
\item fill shaker with ice
\item squeeze lime into shaker glass
  \item add simple syrup
  \item add vodka
 \item shake well and poor
\end{enumerate}
