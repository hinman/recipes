\recipe{Cowboy Salad}
\ingred{
  \begin{itemize}
    \item boneless skinless chicken breast
    \item romaine lettuce
    \item pico de gallo
    \item guacamole
    \item garlic
    \item olive oil
    \item seeded jalapeno peppers
    \item lime juice
    \item fresh ground black pepper
    \item corn
    \item black beans
    \item red pepper
    \item shaved raddish
    \item chihuahua cheese
  \end{itemize}
}

\begin{enumerate}[{\color{blue} 1.}]
\addtolength{\itemindent}{2em}
 \item puree in blender, olive oil, jalapeno peppers, garlic, lime
   juice and pepper
 \item use $\frac{3}{4}$ of marinade on chicken breasts
 \item marinate for at least 1 hour
 \item grill chicken
 \item combine ingredients for salad
 \item use remaining marinade for dressing
\end{enumerate}
