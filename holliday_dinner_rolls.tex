\recipe{Holliday Dinner Rolls}
\ingred{
\begin{itemize}
\item $\frac{1}{3}$ cup water
\item 3 large eggs
\item $\frac{1}{4}$ cup butter cup up
\item 1 tsp salt
\item 3 $\frac{1}{2}$ cups bread flour
\item 3 tbsp sugar
\item 2 tsp bread machine yeast
\end{itemize}
Glaze
\begin{itemize}
\item 1 large egg
\item 1 tbsp sugar
\end{itemize}
}

\begin{enumerate}[{\color{blue} 1.}]
\addtolength{\itemindent}{2em}
 \item Add ingredients to bread machine
 \item Use Dough setting on bread machine
 \item form dough into rosettes
 \item let rise to almost double
 \item apply glaze
 \item base 375 degrees for 12 -- 14 minutes
\end{enumerate}
