\recipe{Dark Rum Mojito}
\ingred{
\begin{itemize}
\item 3 oz. Appleton Dark Rum
\item 1 $\frac{1}{2}$ oz. simple syrup from brown sugar
\item 2 -- 3 oz Club Soda
\item 4 large mint leaves
\item 1 Key Lime, halved
\end{itemize}
}

\begin{enumerate}[{\color{blue} 1.}]
\addtolength{\itemindent}{2em}
 \item squeeze lime into shaker glass, add halves
%
 \item add mint leaves and simple syrup
%
 \item mull with wooden spoon
%
 \item fill with ice
%
 \item add rum and club soda
%
 \item shake well and poor
\end{enumerate}
