\recipe{Shannon's Macaroni and Cheese}
\ingred{
\begin{itemize}
\item 16 oz Jumbo Elbow Macaroni
\item 4 oz Prosciutto
\item 1 pint Heavy Cream
\item 2 cups Parmesan Cheese
\item 2 $\frac{1}{2}$ cups Swiss Cheese
\item 2 cups Milk
\item 1 tsp Nutmeg (Optional)
\end{itemize}
}

\begin{enumerate}[{\color{blue} 1.}]
\addtolength{\itemindent}{2em}
 \item Cook Pasta
 \item Dice Prosciutto
 \item Mix Pasta, Prosciutto, Cream, Milk and Parmesan
 \item Add to 9x13 pan
 \item Top with Salt, Pepper and Swiss Cheese
 \item Bake at 375 degrees for 45 minutes
\end{enumerate}
