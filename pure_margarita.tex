\recipe{Pure Margarita}
\ingred{
\begin{itemize}
\item 1 $\frac{1}{2}$ oz. 100\% agave tequila
\item 1 oz. fresh squeezed lime juice
\item $\frac{3}{4}$ oz. agave nectar
\end{itemize}
}

\begin{enumerate}[{\color{blue} 1.}]
\addtolength{\itemindent}{2em}
 \item Add ice to shaker
%
 \item Add agave nectar, tequila \& lime juice
%
 \item shake
%
 \item Poor into salt rimmed glass (optional) over ice
\end{enumerate}
