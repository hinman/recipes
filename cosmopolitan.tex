\recipe{Cosmopolitan}
\ingred{
\begin{itemize}
\item 1 oz. Vodka
\item $\frac{1}{2}$ oz. fresh squeezed lime juice
\item $\frac{1}{2}$ oz. Triple Sec
\item $\frac{1}{2}$ oz. Cranberry juice
\item $\frac{1}{2}$ tsp. powdered sugar
\end{itemize}
}

\begin{enumerate}[{\color{blue} 1.}]
\addtolength{\itemindent}{2em}
 \item Add ice to shaker
%
 \item mix powdered sugar and lime juice
%
 \item add lime juice, cranberry juice, triple sec and vodka to shaker
%
 \item shake and strain into martini glass
\end{enumerate}
