\recipe{Carrot Cake}
\ingred{
\begin{itemize}
\item 1 $\frac{1}{2}$ cups sugar
\item 1 cup vegetable oil
\item 3 eggs
\item 2 cups all-purpose flour
\item 1 Tbls ground cinnamon
\item 1 tsp baking soda
\item 1 tsp vanilla
\item $\frac{1}{2}$ tsp salt
\item $\frac{1}{4}$ tsp ground nutmeg
\item $\frac{1}{4}$ tsp ground cloves
\item 3 cups coarsely shredded carrots (about 5 medium)
\end{itemize}
}

\begin{enumerate}[{\color{blue} 1.}]
\addtolength{\itemindent}{2em}
 \item Heat oven to 350 degrees
 \item Grease and flour 9x13 inch pan
 \item Mix sugar, oil and eggs in large bowl until blended: beat 1 minute
 \item Stir in remaining ingredients, \emph{\textbf{except carrots}}: beat 1 minute
 \item Stir in carrots
 \item Pour into pan
 \item Bake 35 to 45 minutes or until wooden pick inserted in center
   comes out clean
 \item Cool on wire rack
 \item Frost with cream cheese frosting
\end{enumerate}

\recipe{Cream Cheese Frosting}
\ingred{
\begin{itemize}
\item 8 ounces cream cheese, softened
\item 1 Tbls milk
\item 1 tsp vanilla
\item 4 cups powdered sugar
\end{itemize}
}

\begin{enumerate}[{\color{blue} 1.}]
\addtolength{\itemindent}{2em}
\item Beat cream cheese, milk and vanilla in medium bowl on low speed
  until smooth
\item Gradually beat in powdered sugar, 1 cup at a time until smooth
\end{enumerate}
