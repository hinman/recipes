\recipe{Buttermilk Pancakes}
\ingred{
\begin{itemize}
\item 2 cups all--purpose flour
\item 2 tbsp. sugar
\item 2 tsp. baking powder
\item $\frac{1}{2}$ tsp. baking soda
\item $\frac{1}{2}$ tsp. salt
\item 1 large egg
\item 3 tbsp. unsalted butter, melted
\item 2 cups buttermilk
\item 1 to 2 tsp. vegetable oil
\end{itemize}
}

\begin{enumerate}[{\color{blue} 1.}]
\addtolength{\itemindent}{2em}
 \item Whisk together flour, sugar, baking powder, baking soda and salt
 \item In seperate bowl whisk together egg, melted butter then buttermilk
 \item whisk gently (a few lumps should remain)
 \item cook on 350 degree gridle
\end{enumerate}
